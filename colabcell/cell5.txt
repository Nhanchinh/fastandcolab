from typing import List
import re

def chunk_text_by_sentences(text: str, tokenizer, max_tokens: int = 800) -> List[str]:
    """
    Chia text dài thành các chunks dựa trên ranh giới câu.
    Mỗi chunk ≤ max_tokens, KHÔNG cắt giữa câu để tránh mất ngữ cảnh.
    """
    sentences = re.split(r'(?<=[.!?…])\s+', text)
    sentences = [s.strip() for s in sentences if s.strip()]

    if not sentences:
        return [text]

    total_tokens = len(tokenizer.encode(text, add_special_tokens=False))
    if total_tokens <= max_tokens:
        return [text]

    chunks = []
    current_sentences = []
    current_token_count = 0

    for sent in sentences:
        sent_tokens = len(tokenizer.encode(sent, add_special_tokens=False))

        if current_token_count + sent_tokens > max_tokens and current_sentences:
            chunks.append(" ".join(current_sentences))
            # Giữ câu cuối của chunk trước làm overlap (context bridge)
            last_sent = current_sentences[-1]
            current_sentences = [last_sent]
            current_token_count = len(tokenizer.encode(last_sent, add_special_tokens=False))

        current_sentences.append(sent)
        current_token_count += sent_tokens

    if current_sentences:
        chunks.append(" ".join(current_sentences))

    return chunks if chunks else [text]


def _summarize_single_vit5(text: str, tokenizer, model, max_length: int = 256) -> str:
    """Hàm nội bộ: tóm tắt 1 đoạn ngắn bằng ViT5 (không chunking)"""
    if not text.startswith("summarize:"):
        text = f"summarize: {text}"

    inputs = tokenizer(
        text,
        return_tensors="pt",
        max_length=1024,
        truncation=True
    ).to(device)

    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_length=max_length,
            min_length=30,
            num_beams=6,
            length_penalty=1.2,
            early_stopping=True,
            no_repeat_ngram_size=3,
            repetition_penalty=1.5,
            do_sample=False
        )

    summary = tokenizer.decode(outputs[0], skip_special_tokens=True)

    # Loại bỏ garbage characters
    summary = re.sub(r'[^\w\s.,!?àáảãạăắằẳẵặâấầẩẫậèéẻẽẹêếềểễệìíỉĩịòóỏõọôốồổỗộơớờởỡợùúủũụưứừửữựỳýỷỹỵđÀÁẢÃẠĂẮẰẲẴẶÂẤẦẨẪẬÈÉẺẼẸÊẾỀỂỄỆÌÍỈĨỊÒÓỎÕỌÔỐỒỔỖỘƠỚỜỞỠỢÙÚỦŨỤƯỨỪỬỮỰỲÝỶỸỴĐ]', '', summary)
    summary = re.sub(r'\s+', ' ', summary).strip()

    return summary


def generate_vit5_fin(text: str, max_length: int = 256) -> str:
    """
    Sinh tóm tắt bằng ViT5 Financial v2 - có hỗ trợ Hierarchical Summarization cho text dài.
    
    Cải tiến:
    - Tách chunk theo ranh giới câu (không cắt giữa câu)
    - Per-chunk summary dài hơn (200 tokens) để giữ nhiều thông tin
    - Reduce step chỉ chạy 1 lần (không đệ quy) để tránh mất ngữ cảnh
    """
    token_count = len(vit5_fin_tokenizer.encode(text, add_special_tokens=False))
    
    if token_count <= 800:
        return _summarize_single_vit5(text, vit5_fin_tokenizer, vit5_fin_model, max_length)
    
    print(f"📄 [ViT5-Fin] Text dài ({token_count} tokens) → Chia thành chunks theo câu...")
    
    chunks = chunk_text_by_sentences(text, vit5_fin_tokenizer, max_tokens=800)
    print(f"   📦 Chia thành {len(chunks)} chunks (theo ranh giới câu)")
    
    # Map: Tóm tắt từng chunk với max_length lớn hơn để giữ nhiều thông tin
    per_chunk_length = max(150, max_length // len(chunks) + 80)
    
    chunk_summaries = []
    for i, chunk in enumerate(chunks):
        print(f"   🔄 Đang tóm tắt chunk {i+1}/{len(chunks)} (max {per_chunk_length} tokens)...")
        summary = _summarize_single_vit5(chunk, vit5_fin_tokenizer, vit5_fin_model, max_length=per_chunk_length)
        if summary:
            chunk_summaries.append(summary)
    
    if not chunk_summaries:
        return ""
    
    if len(chunk_summaries) == 1:
        return chunk_summaries[0]
    
    # Reduce: Ghép summaries → tóm tắt lần cuối (1 lần duy nhất, không đệ quy)
    combined = " ".join(chunk_summaries)
    combined_tokens = len(vit5_fin_tokenizer.encode(combined, add_special_tokens=False))
    
    print(f"   🔗 Ghép {len(chunk_summaries)} summaries ({combined_tokens} tokens) → Tóm tắt lần cuối...")
    
    if combined_tokens > 1024:
        # Nếu combined quá dài, cắt bớt thay vì đệ quy
        combined = vit5_fin_tokenizer.decode(
            vit5_fin_tokenizer.encode(combined, add_special_tokens=False)[:900],
            skip_special_tokens=True
        )
    
    final_summary = _summarize_single_vit5(combined, vit5_fin_tokenizer, vit5_fin_model, max_length)
    print(f"   ✅ Hoàn thành tóm tắt hierarchical [ViT5-Fin]!")
    
    return final_summary


def generate_qwen(text: str, max_length: int = 256) -> str:
    """Sinh tóm tắt bằng Qwen2.5-7B"""
    messages = [
        {
            "role": "system",
            "content": (
                "Bạn là một chuyên gia tóm tắt văn bản tiếng Việt. "
                "Quy tắc bắt buộc:\n"
                "1. LUÔN trả lời bằng tiếng Việt. KHÔNG dùng tiếng Trung, tiếng Anh hay ngôn ngữ khác.\n"
                "2. Viết tóm tắt thành MỘT ĐOẠN VĂN liền mạch, tự nhiên, dễ đọc.\n"
                "3. KHÔNG được liệt kê gạch đầu dòng, đánh số 1. 2. 3., hay phân tích theo ý.\n"
                "4. KHÔNG bắt đầu bằng 'Văn bản mô tả về...', 'Bài viết nói về...' hay các cụm giới thiệu tương tự.\n"
                "5. Viết trực tiếp nội dung tóm tắt như đang kể lại câu chuyện một cách ngắn gọn.\n"
                "6. Giữ lại các ý chính quan trọng, bỏ chi tiết phụ, đảm bảo câu văn mạch lạc."
            )
        },
        {
            "role": "user",
            "content": f"Tóm tắt đoạn văn sau thành một đoạn văn ngắn gọn, liền mạch bằng tiếng Việt:\n\n{text}"
        }
    ]

    prompt = qwen_tokenizer.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=True
    )

    inputs = qwen_tokenizer(prompt, return_tensors="pt").to(device)

    outputs = qwen_model.generate(
        **inputs,
        max_new_tokens=max_length,
        do_sample=True,
        temperature=0.5,
        top_p=0.9,
        repetition_penalty=1.2
    )

    full_response = qwen_tokenizer.decode(outputs[0], skip_special_tokens=True)

    if "assistant" in full_response.lower():
        response = full_response.split("assistant")[-1].strip()
    else:
        response = full_response[len(prompt):].strip()

    # Fallback: nếu output chứa quá nhiều ký tự Trung Quốc → retry với temperature thấp hơn
    chinese_chars = len(re.findall(r'[\u4e00-\u9fff]', response))
    total_chars = len(response) if response else 1
    if chinese_chars / total_chars > 0.3:
        print("⚠️ Phát hiện output tiếng Trung, đang retry với temperature=0.1...")
        outputs = qwen_model.generate(
            **inputs,
            max_new_tokens=max_length,
            do_sample=True,
            temperature=0.1,
            top_p=0.95,
            repetition_penalty=1.3
        )
        full_response = qwen_tokenizer.decode(outputs[0], skip_special_tokens=True)
        if "assistant" in full_response.lower():
            response = full_response.split("assistant")[-1].strip()
        else:
            response = full_response[len(prompt):].strip()

    return response


def _split_sentences_finance(text: str) -> list:
    """Tách câu cho extractive summarization (PhoBERT Finance)"""
    t = text.strip()
    if not t:
        return []
    chunks = re.split(r'(?<=[.!?…])\s+|\n+', t)
    return [c.strip() for c in chunks if c.strip()]


def generate_phobert_finance(text: str, max_length: int = 256) -> str:
    """
    Sinh tóm tắt extractive bằng PhoBERT Finance.

    Pipeline:
    1. Tách văn bản thành câu
    2. Encode từng câu qua PhoBERT → CLS embeddings
    3. Inter-Sentence Transformer xử lý context toàn cục
    4. Financial Feature Extractor tạo 6 extra features
    5. Classifier chấm điểm từng câu
    6. Chọn top-k câu quan trọng nhất (giữ thứ tự gốc)
    """
    sents = _split_sentences_finance(text)
    if len(sents) < 2:
        return text.strip()[:500].strip()

    # Xác định số câu cần chọn dựa trên max_length
    # Giữ tỉ lệ nén hợp lý (~30-40% số câu gốc) để thực sự tóm tắt
    if max_length <= 100:
        lo, hi = 2, 3
    elif max_length <= 200:
        lo, hi = 3, 5
    elif max_length <= 350:
        lo, hi = 4, 7
    else:
        lo, hi = 5, 8

    # Với văn bản dài, giới hạn tối đa ~35% số câu để tránh output quá dài
    max_k = max(3, int(len(sents) * 0.35))
    hi = min(hi, max_k)

    k = min(len(sents), max(lo, min(hi, len(sents))))

    n_doc = len(sents)
    n_use = min(n_doc, phobert_fin_max_sents)
    batch_sents = sents[:n_use]

    # Tokenize từng câu
    input_ids = torch.zeros(1, phobert_fin_max_sents, phobert_fin_max_sent_len, dtype=torch.long, device=device)
    attention_mask = torch.zeros(1, phobert_fin_max_sents, phobert_fin_max_sent_len, dtype=torch.long, device=device)
    sentence_mask = torch.zeros(1, phobert_fin_max_sents, dtype=torch.bool, device=device)

    for i, sent in enumerate(batch_sents):
        enc = phobert_fin_tokenizer(
            sent,
            max_length=phobert_fin_max_sent_len,
            padding="max_length",
            truncation=True,
            return_tensors="pt",
        )
        input_ids[0, i] = enc["input_ids"][0].to(device)
        attention_mask[0, i] = enc["attention_mask"][0].to(device)
        sentence_mask[0, i] = True

    # Extra financial features
    extra = None
    if phobert_fin_extra_dim > 0:
        extra = torch.zeros(1, phobert_fin_max_sents, phobert_fin_extra_dim, device=device, dtype=torch.float32)
        mat = _finance_feature_extractor.extract_matrix(batch_sents)
        for i, row in enumerate(mat):
            if i >= phobert_fin_max_sents:
                break
            for j, val in enumerate(row):
                if j < phobert_fin_extra_dim:
                    extra[0, i, j] = float(val)

    # Predict
    with torch.no_grad():
        out = phobert_fin_model.predict(
            input_ids,
            attention_mask,
            sentence_mask,
            extra_features=extra,
            threshold=0.5,
        )
    probs = out["probabilities"][0, :n_use]

    # Chọn top-k câu theo xác suất, giữ thứ tự gốc
    order = probs.argsort(descending=True)
    top_idx = set(order[:k].tolist())
    selected = [sents[i] for i in range(n_doc) if i in top_idx]

    return " ".join(selected)
