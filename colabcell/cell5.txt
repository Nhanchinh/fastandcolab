from typing import List
import re

def chunk_text_by_tokens(text: str, tokenizer, max_tokens: int = 800, overlap_tokens: int = 100) -> List[str]:
    """
    Chia text dài thành các chunks dựa trên số tokens.
    Mỗi chunk ≤ max_tokens, overlap giữa các chunks để không mất ngữ cảnh.
    """
    # Tokenize toàn bộ text
    all_tokens = tokenizer.encode(text, add_special_tokens=False)
    
    if len(all_tokens) <= max_tokens:
        return [text]
    
    chunks = []
    start = 0
    while start < len(all_tokens):
        end = min(start + max_tokens, len(all_tokens))
        chunk_tokens = all_tokens[start:end]
        chunk_text = tokenizer.decode(chunk_tokens, skip_special_tokens=True)
        chunks.append(chunk_text.strip())
        
        # Di chuyển start, trừ overlap để giữ ngữ cảnh
        start = end - overlap_tokens
        if start >= len(all_tokens):
            break
        # Tránh vòng lặp vô hạn nếu overlap >= max_tokens
        if end == len(all_tokens):
            break
    
    return chunks


def _summarize_single_vit5(text: str, tokenizer, model, max_length: int = 256) -> str:
    """Hàm nội bộ: tóm tắt 1 đoạn ngắn bằng ViT5 (không chunking)"""
    if not text.startswith("summarize:"):
        text = f"summarize: {text}"

    inputs = tokenizer(
        text,
        return_tensors="pt",
        max_length=1024,
        truncation=True
    ).to(device)

    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_length=max_length,
            min_length=30,
            num_beams=6,
            length_penalty=1.2,
            early_stopping=True,
            no_repeat_ngram_size=3,
            repetition_penalty=1.5,
            do_sample=False
        )

    summary = tokenizer.decode(outputs[0], skip_special_tokens=True)

    # Loại bỏ garbage characters
    summary = re.sub(r'[^\w\s.,!?àáảãạăắằẳẵặâấầẩẫậèéẻẽẹêếềểễệìíỉĩịòóỏõọôốồổỗộơớờởỡợùúủũụưứừửữựỳýỷỹỵđÀÁẢÃẠĂẮẰẲẴẶÂẤẦẨẪẬÈÉẺẼẸÊẾỀỂỄỆÌÍỈĨỊÒÓỎÕỌÔỐỒỔỖỘƠỚỜỞỠỢÙÚỦŨỤƯỨỪỬỮỰỲÝỶỸỴĐ]', '', summary)
    summary = re.sub(r'\s+', ' ', summary).strip()

    return summary


def generate_vit5(text: str, max_length: int = 256) -> str:
    """
    Sinh tóm tắt bằng ViT5 - có hỗ trợ Hierarchical Summarization cho text dài.
    
    Nếu text ngắn (≤ 800 tokens): tóm tắt trực tiếp
    Nếu text dài (> 800 tokens):
      1. Chia thành chunks (mỗi chunk ≤ 800 tokens, overlap 100 tokens)
      2. Tóm tắt từng chunk
      3. Ghép các tóm tắt → tóm tắt lần cuối (Map-Reduce)
    """
    # Kiểm tra độ dài text
    token_count = len(vit5_tokenizer.encode(text, add_special_tokens=False))
    
    if token_count <= 800:
        # Text ngắn → tóm tắt trực tiếp
        return _summarize_single_vit5(text, vit5_tokenizer, vit5_model, max_length)
    
    # Text dài → Hierarchical Summarization
    print(f"📄 Text dài ({token_count} tokens) → Chia thành chunks để tóm tắt...")
    
    chunks = chunk_text_by_tokens(text, vit5_tokenizer, max_tokens=800, overlap_tokens=100)
    print(f"   📦 Chia thành {len(chunks)} chunks")
    
    # Bước 1 (Map): Tóm tắt từng chunk
    chunk_summaries = []
    for i, chunk in enumerate(chunks):
        print(f"   🔄 Đang tóm tắt chunk {i+1}/{len(chunks)}...")
        summary = _summarize_single_vit5(chunk, vit5_tokenizer, vit5_model, max_length=150)
        if summary:
            chunk_summaries.append(summary)
    
    if not chunk_summaries:
        return ""
    
    # Nếu chỉ có 1 chunk summary → trả luôn
    if len(chunk_summaries) == 1:
        return chunk_summaries[0]
    
    # Bước 2 (Reduce): Ghép summaries → tóm tắt lần cuối
    combined = " ".join(chunk_summaries)
    combined_tokens = len(vit5_tokenizer.encode(combined, add_special_tokens=False))
    
    print(f"   🔗 Ghép {len(chunk_summaries)} summaries ({combined_tokens} tokens) → Tóm tắt lần cuối...")
    
    # Nếu combined vẫn quá dài, đệ quy
    if combined_tokens > 800:
        return generate_vit5(combined, max_length)
    
    final_summary = _summarize_single_vit5(combined, vit5_tokenizer, vit5_model, max_length)
    print(f"   ✅ Hoàn thành tóm tắt hierarchical!")
    
    return final_summary


def score_sentences_with_phobert(sentences: List[str]) -> List[tuple]:
    """Đánh giá độ quan trọng của từng câu bằng PhoBERT"""
    scored = []

    for sent in sentences:
        inputs = phobert_tokenizer(
            sent,
            return_tensors="pt",
            max_length=256,
            truncation=True,
            padding=True
        ).to(device)

        with torch.no_grad():
            outputs = phobert_model(**inputs)
            score = outputs.last_hidden_state.mean().item()

        scored.append((score, sent))

    return scored


def generate_phobert_vit5(sentences: List[str], max_length: int = 256) -> str:
    """Sinh tóm tắt hybrid: PhoBERT ranking + ViT5 generation"""
    if not sentences:
        return ""

    scored = score_sentences_with_phobert(sentences)
    scored.sort(reverse=True, key=lambda x: x[0])

    top_k = max(1, int(len(scored) * 0.6))
    selected = [s for _, s in scored[:top_k]]
    combined_text = " ".join(selected)

    return generate_vit5(combined_text, max_length)


def paraphrase_chunk(text: str, max_length: int = 256) -> str:
    """Paraphrase một chunk text bằng ViT5 Paraphrase"""
    # Thêm prefix "làm mượt: "
    input_text = "làm mượt: " + text.strip()

    inputs = paraphrase_tokenizer(
        input_text,
        return_tensors="pt",
        max_length=256,
        truncation=True
    ).to(device)

    outputs = paraphrase_model.generate(
        **inputs,
        max_length=max_length,
        min_length=20,
        num_beams=5,
        repetition_penalty=1.1,
        early_stopping=True
    )

    result = paraphrase_tokenizer.decode(outputs[0], skip_special_tokens=True)

    # Fix lỗi tokenizer (COVID-19 -> Saxifrag-19)
    result = result.replace("Saxifrag-19", "COVID-19")
    result = result.replace("Saxifrag", "COVID")

    return result.strip()


def generate_phobert_vit5_paraphrase(sentences: List[str], max_length: int = 256, chunk_size: int = 3) -> str:
    """
    Sinh tóm tắt hybrid: PhoBERT extract + ViT5 Paraphrase

    Pipeline:
    1. PhoBERT đánh giá và chọn câu quan trọng
    2. Chia thành chunks (mỗi chunk 3 câu)
    3. ViT5 Paraphrase làm mượt từng chunk với prefix "làm mượt: "
    4. Ghép các chunks lại
    """
    if not sentences:
        return ""

    # Step 1: PhoBERT scoring và ranking
    scored = score_sentences_with_phobert(sentences)
    scored.sort(reverse=True, key=lambda x: x[0])

    # Lấy top 60% câu quan trọng
    top_k = max(1, int(len(scored) * 0.6))
    selected = [s for _, s in scored[:top_k]]

    # Step 2: Chia thành chunks và paraphrase
    paraphrased_parts = []

    for i in range(0, len(selected), chunk_size):
        chunk_sentences = selected[i:i + chunk_size]
        raw_chunk = " ".join(chunk_sentences)

        # Step 3: Paraphrase chunk
        smooth_chunk = paraphrase_chunk(raw_chunk, max_length=max_length)
        paraphrased_parts.append(smooth_chunk)

    # Step 4: Ghép lại
    final_text = " ".join(paraphrased_parts)

    return final_text

def generate_qwen(text: str, max_length: int = 256) -> str:
    """Sinh tóm tắt bằng Qwen2.5-7B"""
    messages = [
        {
            "role": "system",
            "content": (
                "Bạn là một chuyên gia tóm tắt văn bản tiếng Việt. "
                "Quy tắc bắt buộc:\n"
                "1. LUÔN trả lời bằng tiếng Việt. KHÔNG dùng tiếng Trung, tiếng Anh hay ngôn ngữ khác.\n"
                "2. Viết tóm tắt thành MỘT ĐOẠN VĂN liền mạch, tự nhiên, dễ đọc.\n"
                "3. KHÔNG được liệt kê gạch đầu dòng, đánh số 1. 2. 3., hay phân tích theo ý.\n"
                "4. KHÔNG bắt đầu bằng 'Văn bản mô tả về...', 'Bài viết nói về...' hay các cụm giới thiệu tương tự.\n"
                "5. Viết trực tiếp nội dung tóm tắt như đang kể lại câu chuyện một cách ngắn gọn.\n"
                "6. Giữ lại các ý chính quan trọng, bỏ chi tiết phụ, đảm bảo câu văn mạch lạc."
            )
        },
        {
            "role": "user",
            "content": f"Tóm tắt đoạn văn sau thành một đoạn văn ngắn gọn, liền mạch bằng tiếng Việt:\n\n{text}"
        }
    ]

    prompt = qwen_tokenizer.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=True
    )

    inputs = qwen_tokenizer(prompt, return_tensors="pt").to(device)

    outputs = qwen_model.generate(
        **inputs,
        max_new_tokens=max_length,
        do_sample=True,
        temperature=0.5,
        top_p=0.9,
        repetition_penalty=1.2
    )

    full_response = qwen_tokenizer.decode(outputs[0], skip_special_tokens=True)

    if "assistant" in full_response.lower():
        response = full_response.split("assistant")[-1].strip()
    else:
        response = full_response[len(prompt):].strip()

    # Fallback: nếu output chứa quá nhiều ký tự Trung Quốc → retry với temperature thấp hơn
    chinese_chars = len(re.findall(r'[\u4e00-\u9fff]', response))
    total_chars = len(response) if response else 1
    if chinese_chars / total_chars > 0.3:
        print("⚠️ Phát hiện output tiếng Trung, đang retry với temperature=0.1...")
        outputs = qwen_model.generate(
            **inputs,
            max_new_tokens=max_length,
            do_sample=True,
            temperature=0.1,
            top_p=0.95,
            repetition_penalty=1.3
        )
        full_response = qwen_tokenizer.decode(outputs[0], skip_special_tokens=True)
        if "assistant" in full_response.lower():
            response = full_response.split("assistant")[-1].strip()
        else:
            response = full_response[len(prompt):].strip()

    return response
